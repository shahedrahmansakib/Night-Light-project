void setup() {
  Serial.begin(9600);
  pinMode(7, OUTPUT);
}

void loop() {
  int c = analogRead(A0);
  Serial.println(c);

  // Inverted logic
  if (c > 1000) {
    digitalWrite(7, HIGH);   // Dark → LED ON
  } else {
    digitalWrite(7, LOW);    // Light → LED OFF
  }

  delay(200);
}
